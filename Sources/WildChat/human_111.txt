Kotlin Code:

```
import androidx.appcompat.app.AppCompatActivity
import android.os.Bundle
import android.view.View
import android.widget.Button
import android.widget.TextView

class MainActivity : AppCompatActivity() {

    private var result = ""
    private lateinit var resultTextView: TextView

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        resultTextView = findViewById(R.id.result_text)
    }

    fun onNumberClick(v: View) {
        val button = v as Button

        // Add the clicked number to the result string
        result += button.text.toString()

        // Update the result text view
        resultTextView.text = result
    }

    fun onOperationClick(v: View) {
        val button = v as Button

        // Add the operation symbol to the result string
        result += button.text.toString()

        // Update the result text view
        resultTextView.text = result
    }

    fun onClearClick(v: View) {
        // Clear the result string and update the result text view
        result = ""
        resultTextView.text = result
    }

    fun onEqualClick(v: View) {
        // Evaluate the result string and show the result in the text view
        result = eval(result).toString()
        resultTextView.text = result
    }

    private fun eval(s: String): Int {
        // Evaluate the arithmetic expressions and return the result
        return object : Any() {
            var pos = -1
            var ch = 0

            fun nextChar(): Char {
                ch = if (++pos < s.length) s[pos].code else -1
                return ch.toChar()
            }

            fun eat(charToEat: Char): Boolean {
                while (ch == ' '.code) nextChar()
                if (ch == charToEat.code) {
                    nextChar()
                    return true
                }
                return false
            }

            fun parse(): Int {
                var x = parseTerm()
                while (true) {
                    if (eat('+')) x += parseTerm() // addition
                    else if (eat('-')) x -= parseTerm() // subtraction
                    else return x
                }
            }

            fun parseTerm(): Int {
                var x = parseFactor()
                while (true) {
                    if (eat('*')) x *= parseFactor() // multiplication
                    else if (eat('/')) x /= parseFactor() // division
                    else return x
                }
            }

            fun parseFactor(): Int {
                if (eat('(')) {
                    val x = parse()
                    eat(')')
                    return x
                }
                val sb = StringBuilder()
                while (ch in '0'..'9' || ch == '.') {
                    sb.append(nextChar())
                }
                return sb.toString().toInt()
            }
        }.parse()
    }
}
```
