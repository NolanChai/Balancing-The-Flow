*/
    private final Instructions instructions = new Instructions("PromptGenerator") {
      @Override
      public void run() throws Exception {
        String promptTemplate;

        if (getArgument("p").equalsIgnoreCase("v")) {
          promptTemplate = "prompt: --v ";
        } else {
          promptTemplate = "prompt: --ar ";
        }

        String[] prompts = getArgumentsAsString().split(",");
        for (int i = 0; i < prompts.length - 1; i++) {
          // prompts[i] is the description of the image, while prompts[i+1] are environment details
          String promptStr = "";

          if (!prompts[i].equalsIgnoreCase("description") && !prompts[i].equalsIgnoreCase(":")) {
            promptStr += getTemplate().replace("%1%", prompts[i]);
          } else {
            promptStr += "--v";
          }

          if (!prompts[i+1].equalsIgnoreCase("description") && !prompts[i+1].equalsIgnoreCase(":")) {
            if (getTemplate().contains("%2%")) {
              promptStr += getTemplate().replace("%1%", prompts[i]).replace("%2%", prompts[i+1]);
            } else {
              promptStr += getTemplate()
                  .replace(promptTemplate, promptStr)
                  .replace("description", prompts[i])
                  .replace("environment", prompts[i+1]);
            }
          } else {
            if (getTemplate().contains("%2%")) {
              promptStr += getTemplate()
                  .replace("%1%", "--v")
                  .replace("%2%", "5");
            } else {
              promptStr += getTemplate()
                  .replace("%1%", "--v")
                  .replace("description", prompts[i])
                  .replace("environment", "5");
            }
          }
        }
        
        for (int i = 0; i < prompts.length - 2; i++) {
          // prompts[i] is the description of the image, while prompts[i+1] are environment details
          String promptStr = "";
          
          if (!prompts[i].equalsIgnoreCase("description") && !prompts[i].equalsIgnoreCase(":")) {
            promptStr += getTemplate().replace("%1%", prompts[i]);
          } else {
            if (getTemplate().contains("%2%")) {
              promptStr += getTemplate().replace("%1%", prompts[i]).replace("%2%", prompts[i+1]);
            } else {
              promptStr += getTemplate()
                  .replace(promptTemplate, promptStr)
                  .replace("description", prompts[i])
                  .replace("environment", "5");
            }
          }
        }
        
        printLine("Generating 4 distinct prompts with varying descriptions and environments: ", prompts);
      }
    };
