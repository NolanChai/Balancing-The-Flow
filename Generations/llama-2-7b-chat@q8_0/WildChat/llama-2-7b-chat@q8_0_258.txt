```
  // Get the main powerpoint window element
  const powerpointWindow = document.querySelector('.powerpoint-window');

  // Add event listeners for the main powerpoint window
  powerpointWindow.addEventListener('click', (event) => {
    // Prevent the default behavior of the click event
    event.preventDefault();

    // Check if the user clicked inside a slide element
    const isSlideClicked = event.target.tagName === 'SLIDE';

    // If the user clicked inside a slide element, display the next slide
    if (isSlideClicked) {
      displayNextSlide();
    }
  });

  function displayNextSlide() {

    // Get the current slide index
    const currentSlideIndex = powerpointWindow.querySelectorAll('.slide').length;

    // Increment the slide index by 1
    currentSlideIndex += 1;

    // Get the next slide element
    const nextSlide = powerpointWindow.querySelectorAll('.slide')[currentSlideIndex];

    // Display the next slide
    powerpointWindow.replaceChild(nextSlide, powerpointWindow.querySelector('.slide'));
  }

```
